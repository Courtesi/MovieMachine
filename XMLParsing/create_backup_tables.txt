CREATE TABLE IF NOT EXISTS movies_backup (
    id VARCHAR(10) PRIMARY KEY NOT NULL,
    title VARCHAR(100) NOT NULL,
    year INTEGER NOT NULL,
    director VARCHAR(100) NOT NULL
);

CREATE TABLE IF NOT EXISTS stars_backup (
    id VARCHAR(10) PRIMARY KEY NOT NULL,
    name VARCHAR(100) NOT NULL,
    birthYear INTEGER DEFAULT NULL
);

CREATE TABLE IF NOT EXISTS stars_in_movies_backup (
    starId VARCHAR(10) NOT NULL,
    movieId VARCHAR(10) NOT NULL,
    FOREIGN KEY (starId) REFERENCES stars_backup(id),
    FOREIGN KEY (movieId) REFERENCES movies_backup(id),
    PRIMARY KEY (starId, movieId)
);

CREATE TABLE IF NOT EXISTS genres_backup (
    id INTEGER PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(32) NOT NULL
);

CREATE TABLE IF NOT EXISTS genres_in_movies_backup (
    genreId INTEGER NOT NULL,
    movieId VARCHAR(10) NOT NULL,
    FOREIGN KEY (genreId) REFERENCES genres_backup(id),
    FOREIGN KEY (movieId) REFERENCES movies_backup(id),
    PRIMARY KEY (genreId, movieId)
);
